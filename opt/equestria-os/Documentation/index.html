Все 17 приложений разделяют одну схему запуска. Разница только в имени класса окна.

Почему QApplication.setDesktopFileName() важен

Без него KDE не может сопоставить запущенный процесс с .desktop-файлом. Иконка в таскбаре будет "пустой" вместо иконки приложения. Аргумент — имя файла без расширения, например "equestria-os-services-manager".

privilege.py — реальный код из disk-manager / swap-manager
import shutil, subprocess, sys

def find_elevator() -> str | None:
    """Возвращает 'pkexec' или 'kdesu', или None если ничего нет."""
    for tool in ("pkexec", "kdesu"):
        if shutil.which(tool):
            return tool
    return None

def run_privileged(script_path: str, *args) -> subprocess.Popen:
    """Запускает Python-скрипт от имени root через pkexec или kdesu.

    Возвращает Popen-объект — читай stdout построчно:
        for line in proc.stdout:
            if line.startswith("PROGRESS"):
                ...
    """
    elevator = find_elevator()
    if not elevator:
        raise RuntimeError("No privilege escalation tool found (pkexec/kdesu)")

    cmd = [elevator, sys.executable, script_path, *args]
    return subprocess.Popen(
        cmd,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        text=True
    )

import sys

def main():
    # ... получаем args ...
    total = len(sources)

    for i, src in enumerate(sources, 1):
        result = do_operation(src, destination)

        # Протокол: "PROGRESS текущий итого"
        sys.stdout.write(f"PROGRESS {i} {total}\n")
        sys.stdout.flush()   # <-- важно: без flush GUI не получит данные

        if result.error:
            sys.stderr.write(f"ERROR: {result.error}\n")
            sys.exit(1)

    sys.stdout.write(f"OK {total}\n")
    sys.stdout.flush()
    sys.exit(0)

if __name__ == "__main__":
    main()
GUI-сторона: чтение вывода backend в QThread
class BackendWorker(QThread):
    progress = pyqtSignal(int, int)   # current, total
    finished = pyqtSignal(bool, str)  # success, message

    def __init__(self, script, *args):
        super().__init__()
        self.script = script
        self.args = args

    def run(self):
        try:
            proc = privilege.run_privileged(self.script, *self.args)
        except RuntimeError as e:
            self.finished.emit(False, str(e))
            return

        for line in proc.stdout:
            line = line.strip()
            if line.startswith("PROGRESS"):
                _, cur, total = line.split()
                self.progress.emit(int(cur), int(total))
            elif line.startswith("OK"):
                self.finished.emit(True, "")
            elif line.startswith("ERROR"):
                self.finished.emit(False, line[6:])

        proc.wait()
        if proc.returncode != 0 and proc.returncode is not None:
            err = proc.stderr.read()
            self.finished.emit(False, err)

def load_localization_csv(self):
    loc_path = os.path.join(self.base_path, "localization.csv")
    if not os.path.exists(loc_path):
        return

                if i + 1 < len(row)
            }

Общий паттерн: Threading + Signals/Slots

Главное правило PyQt6: нельзя обновлять UI из не-главного потока. Все долгие операции (сеть, диск, pkexec) выносятся в поток, результат передаётся через pyqtSignal.emit().

Паттерн 1: threading.Thread + pyqtSignal (package-manager, services-manager)
from PyQt6.QtCore import pyqtSignal, QObject
import threading

class main_app(QMainWindow):
    # Объявляем сигналы на уровне класса (не в __init__!)
    fetch_finished   = pyqtSignal(list)       # пришёл список пакетов
    uninstall_done   = pyqtSignal(bool, str)  # успех, сообщение

    def __init__(self):
        super().__init__()
        # Подключаем сигналы к слотам (всегда в main thread)
        self.fetch_finished.connect(self.on_packages_loaded)
        self.uninstall_done.connect(self.on_uninstall_result)

        # Запускаем фоновую загрузку
        self.load_packages()

    def load_packages(self):
        def _worker():
            pkgs = self._fetch_all()      # долгая операция
            self.fetch_finished.emit(pkgs)  # безопасно!
        threading.Thread(target=_worker, daemon=True).start()

    def on_packages_loaded(self, packages: list):
        # Этот метод ВСЕГДА вызывается в main thread через event loop
        for pkg in packages:
            self.add_row(pkg)

from PyQt6.QtCore import QThread, pyqtSignal

class AppStoreLoader(QThread):

    """Checkable button showing ☐/☑ — полностью стилизуется через QSS."""

        self._refresh()
        self.stateChanged.emit(2 if checked else 0)  # совместимо с QCheckBox API

    def _refresh(self):
        self.setText(f"☑  {self._label}" if self.isChecked()
                     else f"☐  {self._label}")

    def setChecked(self, val: bool):
        super().setChecked(bool(val))
        self._refresh()

        r = subprocess.run(["pacman", "-Qn"], capture_output=True, text=True)

        # AUR: pacman -Qm (иностранные/AUR)
        r = subprocess.run(["pacman", "-Qm"], capture_output=True, text=True)
        for line in r.stdout.splitlines():
            parts = line.split()
            if parts:
                pkgs.append(PackageData(parts[0], "aur"))

        # Flatpak (если установлен)
        if shutil.which("flatpak"):

                ["flatpak", "list", "--app",
                 "--columns=name,application"],
                capture_output=True, text=True
            )
            for line in r.stdout.splitlines():
                parts = line.split("\t")
                if len(parts) >= 2:
                    pkgs.append(PackageData(parts[0], "flatpak", app_id=parts[1]))

        self.fetch_finished.emit(pkgs)

def detect_root_fstype() -> str:
    """Читает /proc/mounts и возвращает тип ФС корневого раздела."""
    try:
        with open("/proc/mounts") as f:
            for line in f:
                parts = line.split()
                if len(parts) >= 3 and parts[1] == "/":
                    return parts[2].lower()
    except OSError:
        pass
    return "unknown"

def detect_backend() -> str:
    """Возвращает 'btrfs', 'restic', 'timeshift' или 'none'."""
    fs      = detect_root_fstype()
    have_bt = subprocess.run(["which", "btrfs"],
                              capture_output=True).returncode == 0
    have_rs = subprocess.run(["which", "restic"],
                              capture_output=True).returncode == 0
    have_ts = subprocess.run(["which", "timeshift"],
                              capture_output=True).returncode == 0

    if fs == "btrfs" and have_bt:
        return "btrfs"    # нативный CoW: мгновенно, нулевое место
    if have_rs:
        return "restic"   # дедупликация + сжатие
    if have_ts:
        return "timeshift"  # rsync-fallback
    return "none"

def build_hook_apply_script(hook_pacman: bool, hook_flatpak: bool,
                             repo_path: str, move_old_repo: str | None,
                             keep_last: int,
                             restic_repo_default: str) -> str:
    """Возвращает bash-строку для выполнения от root через pkexec."""
    parts = [
        "mkdir -p /var/lib/equestria-save-point",
        f"echo {shlex.quote(str(keep_last))} > {HOOK_CONFIG_PATH}",
    ]
    # Перенос репозитория если путь изменился
    if move_old_repo and move_old_repo != repo_path:
        new_parent = shlex.quote(os.path.dirname(repo_path))
        parts += [
            f"mkdir -p {new_parent}",
            f"mv {shlex.quote(move_old_repo)} {shlex.quote(repo_path)}",
        ]
    # Сохраняем кастомный путь (или удаляем файл если дефолтный)
    if repo_path and repo_path != restic_repo_default:
        parts.append(f"echo {shlex.quote(repo_path)} > {REPO_PATH_SYS}")
    else:
        parts.append(f"rm -f {REPO_PATH_SYS}")

    if hook_pacman:
        parts += [
            "mkdir -p /etc/pacman.d/hooks",
            f"cp {HOOK_SCRIPT_SRC} {PACMAN_HOOK_PATH}",
            f"chmod 644 {PACMAN_HOOK_PATH}",
        ]
    else:
        parts.append(f"rm -f {PACMAN_HOOK_PATH}")

    return " && ".join(parts)

class ServiceData:
    def __init__(self, name: str, unit_file_state: str = "", scope: str = "system"):
        self.name            = name
        self.scope           = scope        # "system" или "user"
        self.description     = ""
        self.active_state    = "inactive"   # active / inactive / failed
        self.sub_state       = "dead"       # running / dead / exited / …
        self.unit_file_state = unit_file_state  # enabled / disabled / static / …

    @property
    def uid(self) -> str:
        return f"{self.scope}:{self.name}"

    @property
    def is_running(self) -> bool:
        return self.active_state == "active"

    @property
    def is_enabled(self) -> bool:
        return self.unit_file_state in ("enabled", "enabled-runtime")

    @property
    def display_name(self) -> str:
        return self.description or self.name.removesuffix(".service")

def _fetch_scope(scope: str) -> dict[str, ServiceData]:
    """Получает все сервисы для scope='system' или 'user'.

    Делает два вызова:
    1. systemctl list-unit-files  → получает unit_file_state (enabled/disabled)
    2. systemctl list-units --all → получает active_state, sub_state, description
    """
    flag = ["--user"] if scope == "user" else ["--system"]
    services = {}

    # Шаг 1: unit-files (включены/выключены)
    r1 = subprocess.run(
        ["systemctl"] + flag + [
            "list-unit-files", "--type=service",
            "--no-pager", "--no-legend"
        ],
        capture_output=True, text=True
    )
    for line in r1.stdout.splitlines():
        parts = line.split()
        if len(parts) >= 2 and parts[0].endswith(".service"):
            services[parts[0]] = ServiceData(
                name=parts[0], unit_file_state=parts[1], scope=scope
            )

    # Шаг 2: units (текущее состояние)
    r2 = subprocess.run(
        ["systemctl"] + flag + [
            "list-units", "--all", "--type=service",
            "--no-pager", "--no-legend", "--full"
        ],
        capture_output=True, text=True
    )
    for line in r2.stdout.splitlines():
        line = line.strip()
        if not line:
            continue
        tokens = line.split(None, 5)
        # Убираем цветовые символы ● ✗ ▷ ○
        if tokens and tokens[0] in ("●", "✗", "▷", "○", "×"):
            tokens = tokens[1:]
        if len(tokens) < 4:
            continue
        name, _, active, sub = tokens[:4]
        desc = tokens[4].strip() if len(tokens) > 4 else ""
        if name in services:
            services[name].active_state = active
            services[name].sub_state = sub
            if desc:
                services[name].description = desc

    return services

CATEGORIES_ORDER = ["system", "software", "appearance"]

def _load_modules(modules_dir: str, t_func, base_path: str) -> list[BaseModule]:
    modules = []
    pattern = os.path.join(modules_dir, "mod_*.py")

    for path in sorted(glob.glob(pattern)):
        mod_name = os.path.basename(path)[:-3]
        spec = importlib.util.spec_from_file_location(mod_name, path)
        mod  = importlib.util.module_from_spec(spec)
        try:
            spec.loader.exec_module(mod)
        except Exception as e:
            print(f"[settings] Failed to load {mod_name}: {e}")
            continue

        # Ищем класс-наследник BaseModule в модуле
        for _, cls in inspect.getmembers(mod, inspect.isclass):
            if issubclass(cls, BaseModule) and cls is not BaseModule:
                instance = cls(t_func, base_path)
                modules.append(instance)
                break

    # Сортируем по категории (CATEGORIES_ORDER) и sort_order внутри категории
    cat_order = {cat: i for i, cat in enumerate(CATEGORIES_ORDER)}
    modules.sort(key=lambda m: (cat_order.get(m.category, 99), m.sort_order))
    return modules

class AppStoreLoader(QThread):
    finished = pyqtSignal(list)

    def run(self):
        pkgs = self._load_with_expac() or self._load_with_pacman()
        self.finished.emit(pkgs)

    def _load_with_expac(self) -> list:
        """expac быстрее и даёт описания."""
        try:
            res = subprocess.run(
                ["expac", "-S", "%n\t%v\t%d\t%r"],
                capture_output=True, text=True
            )
            if res.returncode != 0 or not res.stdout.strip():
                return []
            pkgs = []
            for line in res.stdout.splitlines():
                parts = line.split('\t', 3)
                if len(parts) < 2:
                    continue
                name, version = parts[0], parts[1]
                desc = parts[2].strip() if len(parts) > 2 else ""
                repo = parts[3].strip() if len(parts) > 3 else "pacman"
                d = StoreData(name, version, desc or _GENERIC_PACMAN_DESC, repo)
                d.category = guess_cat(name)
                pkgs.append(d)
            return pkgs
        except FileNotFoundError:
            return []  # expac не установлен → fallback

    def _load_with_pacman(self) -> list:
        pkgs = []
        res = subprocess.run(["pacman", "-Sl"], capture_output=True, text=True)
        for line in res.stdout.splitlines():
            p = line.split()
            if len(p) >= 3:
                d = StoreData(p[1], p[2], _GENERIC_PACMAN_DESC, p[0])
                d.category = guess_cat(p[1])
                pkgs.append(d)
        return pkgs

def _get_ram_gb() -> int:
    """Читает MemTotal из /proc/meminfo → возвращает GB."""
    try:
        with open("/proc/meminfo") as f:
            for line in f:
                if line.startswith("MemTotal:"):
                    kb = int(line.split()[1])
                    return max(1, kb // 1024 // 1024)
    except Exception:
        pass
    return 8  # fallback

class RoundedImageLabel(QLabel):
    """QLabel с закруглёнными краями. Масштабирует изображение динамически."""

    def __init__(self, pixmap: QPixmap, radius: int = 16,
                 max_h: int = 480, parent=None):
        super().__init__(parent)
        self._original = pixmap
        self._radius   = radius
        self.setMaximumHeight(max_h)
        self.setSizePolicy(QSizePolicy.Policy.Expanding,
                           QSizePolicy.Policy.Preferred)

    def hasHeightForWidth(self) -> bool:
        return not self._original.isNull()

    def heightForWidth(self, width: int) -> int:
        """Qt вызывает это при пересчёте layout."""
        if self._original.isNull() or self._original.width() == 0:
            return 0
        aspect = self._original.height() / self._original.width()
        return min(int(width * aspect), self.maximumHeight())

    def paintEvent(self, event):
        if self._original.isNull():
            return
        scaled = self._original.scaled(
            self.width(), self.height(),
            Qt.AspectRatioMode.KeepAspectRatio,
            Qt.TransformationMode.SmoothTransformation
        )
        painter = QPainter(self)
        painter.setRenderHint(QPainter.RenderHint.Antialiasing)
        # Центрируем
        x = (self.width()  - scaled.width())  // 2
        y = (self.height() - scaled.height()) // 2
        # Обрезаем по закруглённому прямоугольнику
        path = QPainterPath()
        path.addRoundedRect(
            float(x), float(y),
            float(scaled.width()), float(scaled.height()),
            float(self._radius), float(self._radius)
        )
        painter.setClipPath(path)
        painter.drawPixmap(x, y, scaled)

def launch_app_from_desktop(self, desktop_name: str):
    """Находит .desktop-файл и запускает приложение."""
    desktop_dirs = [
        "/usr/share/applications/",
        os.path.expanduser("~/.local/share/applications/"),
    ]
    for d in desktop_dirs:
        path = os.path.join(d, desktop_name)
        if os.path.exists(path):
            exec_cmd = self._parse_exec(path)
            if exec_cmd:
                subprocess.Popen(exec_cmd, start_new_session=True)
                return

def _parse_exec(self, desktop_path: str) -> list[str] | None:
    """Читает Exec= из .desktop, убирает %F %U %f %u %i %c %k."""
    with open(desktop_path, encoding="utf-8") as f:
        for line in f:
            if line.startswith("Exec="):
                cmd = line[5:].strip()
                # Убираем placeholder'ы freedesktop
                for ph in ("%F", "%U", "%f", "%u", "%i", "%c", "%k"):
                    cmd = cmd.replace(ph, "")
                return cmd.split()

proton_runner.py — реальный код запуска
APPS_DATA_DIR = os.path.expanduser("~/.local/share/Equestria OS/ProtonApps/")
CONFIG_DIR    = os.path.expanduser("~/.config/Equestria OS/Proton/")

def main():
    exe_path = sys.argv[1]
    exe_name = os.path.basename(exe_path)

    # Уникальный ID приложения: имя + первые 8 символов MD5 пути
    path_hash = hashlib.md5(exe_path.encode("utf-8")).hexdigest()[:8]
    app_id = f"{exe_name}_{path_hash}"

    # Изолированный WINEPREFIX — каждый .exe получает свою "Windows"
    prefix_path = os.path.join(APPS_DATA_DIR, app_id)
    os.makedirs(prefix_path, exist_ok=True)

    # Загрузка пользовательских настроек для этого .exe
    config_file = os.path.join(CONFIG_DIR, f"{app_id}.json")
    settings = {}
    if os.path.exists(config_file):
        with open(config_file) as f:
            settings = json.load(f)

    # Формирование env
    env = os.environ.copy()
    env["WINEPREFIX"] = prefix_path
    env["GAMEID"]     = app_id       # требуется umu-run

    if settings.get("dxvk_hud"):
        env["DXVK_HUD"] = "compiler,frametimes,fps"  # оверлей отладки
    if settings.get("fsr"):
        env["WINE_FULLSCREEN_FSR"] = "1"              # FidelityFX Super Resolution

    extra_args = shlex.split(settings.get("launch_args", ""))
    game_dir   = os.path.dirname(exe_path)

    # Virtual desktop (изолированное окно Wine)
    if settings.get("virtual_desktop"):
        screen = QApplication.primaryScreen().size()
        res = f"{screen.width()}x{screen.height()}"
        cmd = ["umu-run", "explorer.exe",
               f"/desktop=EquestriaOS,{res}", exe_path] + extra_args
    else:
        cmd = ["umu-run", exe_path] + extra_args

    # Запуск с логом
    log_path = os.path.join(APPS_DATA_DIR, f"{app_id}.log")
    log_out  = open(log_path, "w", encoding="utf-8")
    proc = subprocess.Popen(cmd, env=env, cwd=game_dir,
                            stdout=log_out, stderr=subprocess.STDOUT)

    # Splash-окно читает лог и показывает прогресс
    splash = SplashWindow(exe_name, log_path, proc)
    splash.exec()

BaseModule ABC — контракт плагин-системы

Каждый модуль в equestria-os-settings — класс-наследник BaseModule из base_module.py. Settings авто-обнаруживает их через importlib.

base_module.py — реальный полный код
import shutil
from abc import ABC, abstractmethod
from PyQt6.QtWidgets import QWidget


class BaseModule(ABC):
    """Base class for all equestria-os-settings modules.

    Чтобы добавить новый модуль, создайте modules/mod_.py
    с одним классом-наследником BaseModule. Settings авто-обнаружит его.
    """

    # ── Идентификация (class-level атрибуты, задаются в подклассе) ──────────
    module_id:        str = ""   # уникальный ключ, напр. "mod_swap_manager"
    display_name_key: str = ""   # ключ в словаре локализации → название в sidebar
    description_key:  str = ""   # ключ для подзаголовка
    category:         str = "system"  # "system" | "software" | "appearance"
    icon:             str = "⚙"       # emoji в sidebar
    sort_order:       int = 99        # порядок внутри категории (меньше = выше)
    required_binary:  str = ""   # напр. "/usr/bin/btrfs" → is_available() проверяет
    package_name:     str = ""   # напр. "equestria-os-disk-manager" → кнопка Install

    def __init__(self, t_func, base_path: str):
        """
        t_func    — callable(key: str) → str, глобальная t() из SettingsWindow
        base_path — абсолютный путь установки equestria-os-settings
        """
        self.t         = t_func
        self.base_path = base_path
        self._widget: QWidget | None = None

    # ── Lifecycle ────────────────────────────────────────────────────────────

    @abstractmethod
    def build_widget(self) -> QWidget:
        """Строит и возвращает виджет. Вызывается ровно один раз, результат кэшируется."""

    def get_widget(self) -> QWidget:
        """Возвращает кэшированный виджет, строит при первом обращении."""
        if self._widget is None:
            self._widget = self.build_widget()
        return self._widget

    def on_shown(self) -> None:
        """Вызывается каждый раз при переключении на этот модуль.
        Используй для обновления состояния (напр. перечитать swapon --show)."""

    def on_hidden(self) -> None:
        """Вызывается при уходе с модуля. Сохрани несохранённые изменения."""

    def apply_language(self) -> None:
        """Вызывается при смене языка. Перетранслируй все строки виджета."""

    def is_available(self) -> bool:
        """False → модуль скрыт, показывается NotInstalledWidget с кнопкой Install.
        Проверяет наличие required_binary через shutil.which()."""
        if not self.required_binary:
            return True
        return shutil.which(self.required_binary) is not None

    # ── Helpers ──────────────────────────────────────────────────────────────

    def launch_app(self, binary: str) -> None:
        """Запуск внешнего приложения без блокировки UI."""
        import subprocess
        subprocess.Popen([binary], start_new_session=True)

    def launch_in_terminal(self, cmd: list) -> None:
        """Запуск команды в первом доступном терминале."""
        import subprocess
        for term_args in [
            ["konsole", "-e"],
            ["xterm", "-e"],
            ["gnome-terminal", "--"],
        ]:
            if shutil.which(term_args[0]):
                subprocess.Popen(term_args + cmd, start_new_session=True)
                return

Типичные ошибки PyQt6 и их диагностика
Обновление UI из потока: RuntimeError: Internal C++ object already deleted или крэш без сообщения → всегда emit() сигнал, никогда не трогать виджет напрямую из потока.

QThread не завершается: приложение зависает при закрытии → добавить self.thread.quit(); self.thread.wait() в closeEvent().

QSS не применяется: Kvantum перекрывает стили → QApplication.setStyle("Fusion") перед setStyleSheet(). Или: забыли вызвать widget.style().unpolish(); widget.style().polish() после setProperty().

Иконка в таскбаре пустая: забыли QApplication.setDesktopFileName("app-name") — должно совпадать с именем .desktop-файла без расширения.

Шрифт не загружается: QFontDatabase.applicationFontFamilies(fid) возвращает пустой список → файл .ttf не найден или повреждён. Проверить путь: os.path.exists(f_path).

pkexec не показывает диалог: нет .policy-файла в /usr/share/polkit-1/actions/, или запускается из нет-терминала без DISPLAY/WAYLAND_DISPLAY. Проверить: echo $XDG_CURRENT_DESKTOP.

subprocess зависает: PIPE без чтения буферизуется и блокирует дочерний процесс → использовать communicate() или читать stdout построчно в потоке. Никогда не proc.wait() пока PIPE не прочитан.

QDialog не показывается поверх: добавить Qt.WindowType.WindowStaysOnTopHint в setWindowFlags() — используется в git-askpass и SplashWindow proton.
